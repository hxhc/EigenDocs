139template<typename Scalar> struct swap_assign_op {
140
141 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void assignCoeff(Scalar& a, const Scalar& b) const
142 {
143#ifdef EIGEN_GPUCC
144 // FIXME is there some kind of cuda::swap?
145 Scalar t=b; const_cast<Scalar&>(b)=a; a=t;
146#else
147 using std::swap;
148 swap(a,const_cast<Scalar&>(b));
149#endif
150 }
151};
152template<typename Scalar>
153struct functor_traits<swap_assign_op<Scalar> > {
154 enum {
155 Cost = 3 * NumTraits<Scalar>::ReadCost,
156 PacketAccess =
157 #if defined(EIGEN_VECTORIZE_AVX) && (EIGEN_CLANG_STRICT_LESS_THAN(8,0,0) || EIGEN_COMP_CLANGAPPLE)
158 // This is a partial workaround for a bug in clang generating bad code
159 // when mixing 256/512 bits loads and 128 bits moves.
160 // See http://eigen.tuxfamily.org/bz/show_bug.cgi?id=1684
161 // https://bugs.llvm.org/show_bug.cgi?id=40815
162 0
163 #else
164 packet_traits<Scalar>::Vectorizable
165 #endif
166 };
