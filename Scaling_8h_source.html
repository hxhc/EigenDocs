83 operator* (const Transform<Scalar, Dim, Mode, Options>& t) const
84 {

180template<typename Scalar>
181template<int Dim>
182inline Transform<Scalar,Dim,Affine>
183UniformScaling<Scalar>::operator* (const Translation<Scalar,Dim>& t) const
184{
185 Transform<Scalar,Dim,Affine> res;
186 res.matrix().setZero();
187 res.linear().diagonal().fill(factor());
188 res.translation() = factor() * t.vector();
189 res(Dim,Dim) = Scalar(1);
190 return res;
191}
